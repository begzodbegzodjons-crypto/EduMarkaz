   813→function SimpleBarChart({ data }: { data: { label: string; value: number }[] }) {
   814→  const max = Math.max(...data.map((d) => d.value), 1)
   815→  return (
   816→    <div className="flex items-end gap-2 h-40">

   818→        <div key={i} className="flex-1 flex flex-col items-center gap-1.5">
   819→          <div className="text-[10px] text-muted-foreground font-medium">
   820→            {d.value > 0 ? Math.round(d.value / 1000) + 'k' : ''}
   821→          </div>
   822→          <div className="w-full bg-muted rounded-t-lg overflow-hidden flex items-end" style={{ height: '100%' }}>
   823→            <motion.div
   824→              initial={{ height: 0 }}
   825→              animate={{ height: `${(d.value / max) * 100}%` }}
   826→              transition={{ duration: 0.6, delay: i * 0.1 }}
   827→              className="w-full bg-gradient-to-t from-emerald-500 to-teal-400 rounded-t-lg"
   828→            />
   829→          </div>
   830→          <div className="text-[10px] text-muted-foreground">{d.label}</div>
   831→        </div>
   832→      ))}
   833→    </div>
